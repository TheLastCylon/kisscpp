330 //--------------------------------------------------------------------------------
331 // Used to implement stream manipulators that are called (with arguments) by the user as part of the output operation.
332 // E.g.:
333 // ss << via::vmlog::src("MyLoggingFunction") << via::vmlog::endl;
334 //
335 // What makes these cases unique is that @c operator<< is being passed the
336 // result of a function call, as opposed to a function pointer, as in the case
337 // of something such as via::vmlog::endl:
338 // ss << "Testing, 1 2 3" << via::vmlog::endl;
339 // T - The manipulator's argument type.
340 template<typename T> class ManipResult
341 {
342  public:
343  typedef LogStream& (*func_type) (LogStream&, T); // The function type.
344 
345  ManipResult(func_type f, T t) : mFunc(f), mT(t) {} // f - The function.
346  // t - The manipulator's parameter. Stored until operator<< calls operator(), which will pass in the stream.
347 
348  LogStream& operator()(LogStream& s) { return mFunc(s, mT); } // Calls the stored function, passing into it its stream argument and the stored parameter.
349  private:
350  func_type mFunc; // The function.
351  T mT; // The parameter that will be passed into @a func.
352 };
