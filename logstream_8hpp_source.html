298  //--------------------------------------------------------------------------------
299  // What makes these cases unique is that insertion operator is being passed the
300  // result of a function call, as opposed to a function pointer
301  // ss << "Testing, 1 2 3" << kisscpp::manip::endl;
302  // T - The manipulator's argument type.
303  template<typename T> class ManipResult
304  {
305  public:
306  typedef LogStream& (*func_type) (LogStream&, T); // The function type.
307 
308  ManipResult(func_type f, T t) : mFunc(f), mT(t) {} // f - The function.
309  // t - The manipulator's parameter.
310  // Stored until operator<< calls operator(), which will pass in the stream.
311 
312  LogStream& operator()(LogStream& s) { return mFunc(s, mT); } // Calls the stored function, passing into it its stream argument and the stored parameter.
313  private:
314  func_type mFunc; // The function.
315  T mT; // The parameter that will be passed into @a func.
316  };
